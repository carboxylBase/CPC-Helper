name: Release
on:
  push:
    tags:
      - 'v*'

jobs:
  publish-tauri:
    permissions:
      contents: write
    strategy:
      fail-fast: false
      matrix:
        platform: [macos-latest, ubuntu-22.04, windows-latest]

    runs-on: ${{ matrix.platform }}
    steps:
      - uses: actions/checkout@v4

      - name: setup node
        uses: actions/setup-node@v4
        with:
          node-version: lts/*

      - name: install Rust stable
        uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.platform == 'macos-latest' && 'aarch64-apple-darwin,x86_64-apple-darwin' || '' }}

      - name: install dependencies (ubuntu only)
        if: matrix.platform == 'ubuntu-22.04'
        run: |
          sudo apt-get update
          sudo apt-get install -y libwebkit2gtk-4.1-dev libappindicator3-dev librsvg2-dev patchelf

      - name: install frontend dependencies
        run: npm install

      - uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          TAURI_SIGNING_PRIVATE_KEY: ${{ secrets.TAURI_SIGNING_PRIVATE_KEY }}
          TAURI_SIGNING_PRIVATE_KEY_PASSWORD: ${{ secrets.TAURI_SIGNING_PRIVATE_KEY_PASSWORD }}
        with:
          tagName: v__VERSION__
          releaseName: 'CPC Helper v__VERSION__'
          releaseBody: 'See the assets to download this version and install.'
          releaseDraft: true
          prerelease: false

  generate-updater-json:
    needs: publish-tauri
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/checkout@v4

      - name: setup node
        uses: actions/setup-node@v4
        with:
          node-version: lts/*

      - name: Generate latest.json
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            
            // 1. 获取版本信息
            const packageJson = JSON.parse(fs.readFileSync('package.json', 'utf8'));
            const version = packageJson.version;
            const tagName = `v${version}`;
            
            console.log(`Processing update manifest for: ${tagName}`);

            // 2. 获取 Release 资产信息
            const releases = await github.rest.repos.listReleases({
              owner: context.repo.owner,
              repo: context.repo.repo,
            });
            const release = releases.data.find(r => r.tag_name === tagName);
            
            if (!release) {
              core.setFailed(`Release not found for tag ${tagName}`);
              return;
            }

            // [调试日志] 打印当前 Release 里的所有文件
            console.log("Current assets in release:", release.assets.map(a => a.name));

            // 3. [关键修复] 检查并删除已存在的 latest.json，防止 'already_exists' 报错
            const existingManifest = release.assets.find(a => a.name === 'latest.json');
            if (existingManifest) {
              console.log(`Deleting existing latest.json (id: ${existingManifest.id}) to avoid conflict...`);
              await github.rest.repos.deleteReleaseAsset({
                owner: context.repo.owner,
                repo: context.repo.repo,
                asset_id: existingManifest.id,
              });
            }

            const updateData = {
              version: version,
              notes: release.body || "Update available",
              pub_date: release.published_at || new Date().toISOString(),
              platforms: {}
            };

            // 4. 平台匹配逻辑：支持 Windows 多 Key
            function getPlatformKeys(filename) {
              const lower = filename.toLowerCase();
              if (lower.endsWith('.app.tar.gz')) {
                return lower.includes('aarch64') ? ['darwin-aarch64'] : ['darwin-x86_64'];
              }
              if (lower.endsWith('.appimage.tar.gz') || lower.endsWith('.appimage')) {
                return ['linux-x86_64'];
              }
              // 关键：同时匹配 exe 和 nsis.zip，并生成两个 key
              if (lower.endsWith('.msi.zip') || lower.endsWith('.exe') || lower.endsWith('.nsis.zip')) {
                return ['windows-x86_64', 'windows-x86_64-nsis'];
              }
              return [];
            }

            const assets = release.assets;
            let foundSig = false;

            // 5. 遍历资产，下载签名并填充 JSON
            for (const asset of assets) {
              if (asset.name.endsWith('.sig')) {
                foundSig = true;
                const binaryName = asset.name.slice(0, -4); // 去掉 .sig
                const binaryAsset = assets.find(a => a.name === binaryName);
                
                if (binaryAsset) {
                  const keys = getPlatformKeys(binaryName);
                  if (keys.length > 0) {
                    // console.log(`[Match] ${binaryName} -> ${keys.join(', ')}`);
                    const signature = await fetch(asset.browser_download_url).then(res => res.text());
                    for (const key of keys) {
                      updateData.platforms[key] = {
                        signature: signature.trim(),
                        url: binaryAsset.browser_download_url
                      };
                    }
                  }
                }
              }
            }

            if (!foundSig) {
              console.log("⚠️ No .sig files found! Updates will fail if binaries are not signed.");
            }

            // 6. 写入文件并上传
            console.log("Final Manifest content:", JSON.stringify(updateData, null, 2));
            fs.writeFileSync('latest.json', JSON.stringify(updateData, null, 2));

            await github.rest.repos.uploadReleaseAsset({
              owner: context.repo.owner,
              repo: context.repo.repo,
              release_id: release.id,
              name: 'latest.json',
              data: fs.readFileSync('latest.json')
            });
